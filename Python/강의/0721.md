```python
import json # 내장되어있는 모듈. why json 설치 없이 씀

#json 데이터
json_data = '''
{
    "name": "김싸피",
    "age": 28
    "hobbies": [
        "공부하기",
        "복습하기"
    ]
}
'''


data = json.loads(json_data) #dictionary로 변경
print(data)

# json 데이터에서 원하는 데이터만 가져오기
name = data.get('name')

print(name)

# name이라는 데이터만 가져오고 싶다. -> parsing
# pasing -> 원하는 형태로 변환
```
